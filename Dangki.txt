<script>
  (function(){
    const tabs = document.querySelectorAll('.kzh-tab');
    const panels = document.querySelectorAll('[data-langpanel]');

    function setActive(lang){
      tabs.forEach(t=>{
        const is = t.getAttribute('data-lang') === lang;
        t.classList.toggle('active', is);
        t.setAttribute('aria-selected', is? 'true' : 'false');
      });
      panels.forEach(p=>{
        p.style.display = p.getAttribute('data-langpanel') === lang ? '' : 'none';
      });
      // update subtitle
      const sub = document.getElementById('kzh-sub');
      if(lang === 'vi') sub.textContent = 'Đây là những chức năng sẽ được cập nhật cho tuần sau · Các chức năng mới và sửa một số chức năng cũ';
      else sub.textContent = 'These features will be updated next week. New features and fixes for some existing features.';
    }

    tabs.forEach(t => t.addEventListener('click', ()=> setActive(t.getAttribute('data-lang'))));

    // Optional: remember preference in session
    const saved = sessionStorage.getItem('kzh-lang');
    if(saved) setActive(saved);
    tabs.forEach(t => t.addEventListener('click', ()=> sessionStorage.setItem('kzh-lang', t.getAttribute('data-lang'))));

    // Accessibility: close via Escape
    document.addEventListener('keydown', e => {
      if(e.key === 'Escape'){
        const el = document.getElementById('kzh-notice');
        if(el) el.style.display = 'none';
      }
    });
  })();
</script>
